#### 1
```cpp
class Solution {
public:
    int rob(vector<int>& nums) {
        int size=nums.size();
        if(!size){return 0;}
        int dp[105]={0};
        dp[1]=nums[0];
        for(int i=2;i<=size;i++){
            dp[i]=max(dp[i-1],dp[i-2]+nums[i-1]);
        }
        return dp[size];
    }
};
```

```cpp
class Solution {
public:
    int rob(vector<int>& nums) {
        int size=nums.size();
        if(!size){return 0;}
        if(size==1){return nums[0];}
        int first = nums[0], second = max(nums[0],nums[1]), temp;
        //优化空间复杂度
        //实际上在leetcode上使用的空间和未优化的差不多
        for(int i=2; i<size; i++){
            temp = second;
            second = max(first+nums[i],second);
            first = temp;
        }
        return second;
    }
};
```
